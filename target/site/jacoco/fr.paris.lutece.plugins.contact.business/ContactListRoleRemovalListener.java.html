package fr.paris.lutece.plugins.contact.business;

import fr.paris.lutece.plugins.contact.service.ContactPlugin;
import fr.paris.lutece.portal.service.i18n.I18nService;
import fr.paris.lutece.portal.service.plugin.PluginService;
import fr.paris.lutece.portal.service.util.RemovalListener;

import java.util.Collection;
import java.util.Locale;

/**
 * ContactList Removal Listener
 */
public class ContactListRoleRemovalListener implements RemovalListener
{
    private static final String PROPERTY_ROLE_CANNOT_BE_REMOVED = "contact.message.roleCannotBeRemoved";

    /**
     * Check if the object can be safely removed
     * 
     * @param strId
     *            The object id
     * @return true if the object can be removed otherwise false
     */
    public boolean canBeRemoved( String strId )
    {
        if ( strId == null )
        {
            return true;
        }

        Collection<ContactList> listContactList = ContactListHome.findByRoleKey( strId, PluginService.getPlugin( ContactPlugin.PLUGIN_NAME ) );

        if ( ( listContactList != null ) && ( listContactList.size( ) > 0 ) )
        {
            return false;
        }

        return true;
    }

    /**
     * Gives a message explaining why the object can't be removed
     * 
     * @param strId
     *            The object id
     * @param locale
     *            The current locale
     * @return The message
     */
    public String getRemovalRefusedMessage( String strId, Locale locale )
    {
        // Build a message contact for using this role Lutece
        return I18nService.getLocalizedString( PROPERTY_ROLE_CANNOT_BE_REMOVED, locale );
    }
}
